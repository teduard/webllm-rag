// Main-thread interface to the embedder Web Worker.
// Uses the official HuggingFace pattern: plain .js worker, singleton pipeline.

type WorkerMsg =
  | { type: "ready" }
  | { type: "result"; embeddings: number[][]; id: number }
  | { type: "error"; message: string; id: number }
  | { type: "progress"; [key: string]: unknown };

let worker: Worker | null = null;
let workerReady = false;
const readyCallbacks: Array<() => void> = [];
const pending = new Map<number, {
  resolve: (v: number[][]) => void;
  reject: (e: Error) => void;
}>();
let callId = 0;

function getWorker(): Promise<void> {
  return new Promise((resolve, reject) => {
    if (workerReady) { resolve(); return; }

    if (!worker) {
      // NOTE: .js extension - plain JS worker, not processed by TypeScript/Vite bundler
      worker = new Worker(
        new URL("../workers/embedder.worker.js", import.meta.url),
        { type: "module" }
      );

      worker.addEventListener("message", (e: MessageEvent<WorkerMsg>) => {
        const msg = e.data;
        if (msg.type === "ready") {
          workerReady = true;
          readyCallbacks.forEach((cb) => cb());
          readyCallbacks.length = 0;
        } else if (msg.type === "result") {
          pending.get(msg.id)?.resolve(msg.embeddings);
          pending.delete(msg.id);
        } else if (msg.type === "error") {
          const err = new Error(msg.message);
          if (msg.id === -1) {
            readyCallbacks.length = 0;
            reject(err);
          } else {
            pending.get(msg.id)?.reject(err);
            pending.delete(msg.id);
          }
        }
        // ignore "progress" messages
      });

      worker.addEventListener("error", (e) => {
        reject(new Error(e.message ?? "Worker error"));
      });
    }

    readyCallbacks.push(resolve);
  });
}

export async function embed(texts: string[]): Promise<number[][]> {
  await getWorker();
  const id = callId++;
  return new Promise((resolve, reject) => {
    pending.set(id, { resolve, reject });
    worker!.postMessage({ type: "embed", texts, id });
  });
}

export async function embedOne(text: string): Promise<number[]> {
  const [vec] = await embed([text]);
  return vec;
}
